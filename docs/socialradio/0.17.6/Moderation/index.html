    Moderation¶

    SocialRadio offers moderation of content by allowing the user to report inapropriate mixtapes or users. To do so, you use

    • reportUser
    • reportMixtape
    Reporting inapropriate content
    class MyApplicationActivity : AppCompatActivity() {

    ...
    fun onReportUserButtonPressed(identifier: String) {
        SocialRadio.getInstance()?.reportUser(identifier, onComplete = {
        },
        onError = { error, status -> 
            Log.i("MyLog", "RadioSession failed to report a user")
        })
    }

    fun onReportMixtapeButtonPressed(identifier: String) {
        SocialRadio.getInstance()?.reportMixtape(identifier, onComplete = {
        },
        onError = { error, status -> 
            Log.i("MyLog", "RadioSession failed to report a user")
        })
    }
    ...

}

    SocialRadio also offers the possibility of sending users a notification. To do so, you use

    • notifyFollowers
    Notifying users
    class MyApplicationActivity : AppCompatActivity() {

    ...
    fun notifyFollowers(identifier: String, title: String, body: String, objectType: String?, objectId: String?) {
        SocialRadio.getInstance()?.notifyFollowers(userIdentifier = identifier, title, body, objectType, objectId, onComplete = {
        },
        onError = { error -> 
            Log.i("MyLog", "RadioSession failed to notify users")
        })
    }
    ...

}
