SocialLayerListeners

class SocialLayerListeners(var authenticationListener: SocialLayerAuthenticationListener, var userListener: SocialLayerUserListener, var liveMixtapeListener: SocialLayerLiveMixtapeListener, var mixtapeLikesListener: SocialLayerMixtapeLikesListener? = null, var mixtapeCommentsListener: SocialLayerMixtapeCommentsListener? = null, var mixtapeBackgroundsListener: SocialLayerMixtapeBackgroundsListener? = null, var trackLikesListener: SocialLayerTrackLikesListener? = null, var moderationListener: SocialLayerModerationListener? = null, var notificationsListener: SocialLayerNotificationsListener? = null)

SocialLayerDelegates is an object that holds all delegates used to communicate between the SocialRadio SDK and a backend that handles storage of social interactions. 'authenticationDelegate', 'userDelegate' and 'liveMixtapeDelegate' are required always. Every other delegate can be nil and is only required if your backend and SocialRadio implementation supports the attached features, like commenting, moderation, backgrounds, etc. See each delegate's documentation for details.
